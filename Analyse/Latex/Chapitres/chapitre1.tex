\exemples{On ÃĐtudie l'ÃĐxistence de l'intÃĐgrale gÃĐnÃĐralisÃĐe

	F(t) = \int_t^1 \frac1x \mathrm d x = \left. \ln x \right|_t^1 = - \ln t

	On en conclut alors que

\newlem{Soit $f, g : [a,b[ \to \R$ deux fonctions intÃĐgrables sur tout sous-intervalle $J \subset [a,b[$ et supposons qu'il existe $c \in [a,b[$ tel que,

\newdef{Soit $f : ]a,b[ \to \R$ intÃĐgrable sur tout sous-intervalle fermÃĐ $J \subset ]a,b[$. Pour $c \in ]a,b[$, si les deux intÃĐgrales gÃĐnÃĐralisÃĐes

\newprop{Soit $f : ]a,b[ \to \R$ intÃĐgrable sur tout sous-intervalle fermÃĐ $J \subset ]a,b[$ alors l'existence (ou non) de l'intÃĐgrale gÃĐnÃĐralisÃĐe $\int_a^b f(x) ~\mathrm dx$ ne dÃĐpend pas du choix de $c \in ]a,b[$.

\prf{Soit $\tilde c \in ]a,b[, \tilde c \neq x$, alors
	\[
	\int_a^{\tilde c} f(x) ~\mathrm dx = \int_a^c f(x) ~\mathrm dx + \int_c^{\tilde c}
	\]
	existe si et seulement si $\int_a^c f(x) ~\mathrm dx$ existe, car $f$ est intÃĐgrale sur $[c, \tilde c]\cup[\tilde c,c]$. De mÃŠme,
	\[
	\int_{\tilde c}^b f(x) ~\mathrm dx = \int_{\tilde c}^c f(x) ~\mathrm dx + \int_c^b f(x) ~\mathrm dx
	\]
	existe si et seulement si $\int_c^b f(x) ~\mathrm dx$ existe. Et 
	\[
	\int_a^{\tilde c} f(x) ~\mathrm dx + \int_{\tilde c}^b f(x) ~\mathrm dx =  \int_a^c f(x) ~\mathrm dx + \int_c^{\tilde c} f(x) ~\mathrm dx  + \int_{\tilde c}^c f(x) ~\mathrm dx  + \int_c^b f(x) ~\mathrm dx  = \int_a^b f(x) ~\mathrm dx

\newdef{Soit $I$ un intervalle bornÃĐ quelconque et $f : I \to \R$ intÃĐgrable sur tout sous-intervalle fermÃĐ bornÃĐe. On dit que l'intÃĐgrale gÃĐnÃĐralisÃĐe \emph{converge absolument} si

\newprop{Si l'intÃĐgrale $\int_I f(x) ~\mathrm dx$ est absolument convergente, alors elle converge.}{}

\prf{On pose $f_+(x) = \max\{ f(x), 0 \}$ et $f_-(x) = -\min \{ f(x),0\}$. On a par dÃĐfinition $f(x) = f_+(x)  - f_-(x)$ et $|f(x)| = f(x)_+ + f_-(x)$, ainsi

}

\subsection{IntÃĐgrale gÃĐnÃĐralisÃĐe sur un intervalle non bornÃĐ}

\newdef{Si $I = [a, \infty[$ et $f : I \to \R$ intÃĐgrable sur tout sous intervalle fermÃĐ bornÃĐ. Soit $F(t) = \int_a^t f(x) \mathrm d x$, si $\lim_{t \to \infty} F(t)$ existe alors on dit que l'intÃĐgrale gÃĐnÃĐralisÃĐe $\int_a^\infty f(x) \mathrm d x$ converge et on note

	\[
	\int_a^\infty f(x) ~\mathrm d x = \int_a^c f(x) ~\mathrm d x + \int_c^\infty f(x) ~\mathrm d x

\exemples{Soit $f(x) = \frac1{x^\alpha}$
